---
title: Signup
api: POST /auth/signup
description: Crear una nueva cuenta de usuario
---

<ParamField body="email" type="string" required>
  Correo electrónico del nuevo usuario
</ParamField>

<ParamField body="password" type="string" required>
  Contraseña (mínimo 8 caracteres)
</ParamField>

<ParamField body="first_name" type="string" required>
  Nombre
</ParamField>

<ParamField body="last_name" type="string" required>
  Apellido
</ParamField>

<ParamField body="workspace_id" type="number">
  ID del workspace al que pertenece (opcional en registro inicial)
</ParamField>

<ParamField body="role_id" type="number">
  ID del rol asignado
</ParamField>

<RequestExample>
```bash cURL
curl -X POST https://api.scrivot.cl/auth/signup \
  -H "Content-Type: application/json" \
  -d '{
    "email": "nuevo@empresa.cl",
    "password": "miContraseña123",
    "first_name": "Juan",
    "last_name": "Pérez"
  }'
```
</RequestExample>
